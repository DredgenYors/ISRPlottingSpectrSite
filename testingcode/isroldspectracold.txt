    <script>
        const defaultValues = {
            // Non-constant values
            nu_i: "1.0e-7",
            nu_e: "1.0e-7",
            ni: "2.0e11",
            ne: "2.0e11",
            theta: "60",
            Te: "500",
            n_terms: "2000",
            
            // Constant values
            mi: "2.65686e-26",
            me: "9.11e-31",
            B: "3.6e-5",
            epsilon_0: "8.854187817e-12",
            kB: "1.380649e-23",
            e: "1.602e-19"
        };

        document.addEventListener("DOMContentLoaded", function() {
            if (typeof Plotly === 'undefined') {
                console.error("Plotly failed to load.");
            } else {
                generatePlot();
            }
        });

        function generatePlot() {
            let newValues = {};
            document.querySelectorAll("input").forEach(input => {
                newValues[input.id] = parseFloat(input.value);
            });

            fetch('/update_values', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(newValues)
            })
            .then(response => response.json())
            .then(() => fetch('/plot'))
            .then(response => response.json())
            .then(data => {
                console.log("Received Data:", data);
                if (!data.frequencies || !data.spectra) {
                    console.error("Error: Missing data for plot");
                    return;
                }
                
                let trace = {
                    x: data.frequencies,
                    y: data.spectra,
                    mode: 'lines',
                    name: 'ISR Spectra'
                };
                let layout = {
                    title: 'ISR Spectra for Different Electron Temperatures',
                    xaxis: { title: 'Frequency (MHz)' },
                    yaxis: { title: 'Spectra', type: 'log' },
                    hovermode: 'x unified',
                    autosize: true
                };
                Plotly.newPlot('plot', [trace], layout, { responsive: true });
            })
            .catch(error => console.error("Error generating plot:", error));
        }

        function resetValues() {
            document.querySelectorAll("input").forEach(input => {
                input.value = input.defaultValue;
            });
            generatePlot();
        }
    </script>
